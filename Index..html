<script>

// --------------------------------------------------------------

// 30 x LEVEL 2 FOOD HYGIENE QUESTIONS (UK standards - CIEH/RSPH aligned)

const QUESTION_BANK = [

{ text: "What is the temperature range of the 'danger zone' for bacterial growth?", options: ["0°C – 10°C", "5°C – 63°C", "8°C – 68°C", "Below 5°C"], correct: 1 },

];

// make sure exactly 30 questions

console.assert(QUESTION_BANK.length === 30, "Need 30 questions, found: " + QUESTION_BANK.length);

// ----- helper functions -----

let currentMode = "training"; // 'training' or 'exam'

// training state

let trainingIndex = 0;

let trainingAnswers = new Array(QUESTION_BANK.length).fill(null); // store selected option index (0-3)

// exam state

let examIndex = 0;

let examAnswers = new Array(QUESTION_BANK.length).fill(null);

let examSubmitted = false;

let examTimer = null;

let examTimeSeconds = 18 * 60; // 18 minutes for 30 questions (reasonable)

// DOM elements

const trainingSection = document.getElementById('trainingSection');

const examSection = document.getElementById('examSection');

const trainingModeBtn = document.getElementById('trainingModeBtn');

const examModeBtn = document.getElementById('examModeBtn');

const trainingProgressSpan = document.getElementById('trainingProgress');

const trainingQuestionDiv = document.getElementById('trainingQuestion');

const trainingOptionsDiv = document.getElementById('trainingOptions');

const trainingFeedbackDiv = document.getElementById('trainingFeedback');

const trainingNextBtn = document.getElementById('trainingNextBtn');

const examProgressSpan = document.getElementById('examProgress');

const examQuestionDiv = document.getElementById('examQuestion');

const examOptionsDiv = document.getElementById('examOptions');

const examFeedbackDiv = document.getElementById('examFeedback');

const examNextBtn = document.getElementById('examNextBtn');

const submitExamBtn = document.getElementById('submitExamBtn');

// ----- render training current question -----

function renderTraining() {

const q = QUESTION_BANK[trainingIndex];

trainingProgressSpan.innerText = `Question ${trainingIndex+1} / 30`;

trainingQuestionDiv.innerText = q.text;

// render options

trainingOptionsDiv.innerHTML = '';

q.options.forEach((opt, idx) => {

const optionDiv = document.createElement('div');

optionDiv.className = 'option';

if (trainingAnswers[trainingIndex] === idx) optionDiv.classList.add('selected');

optionDiv.innerText = `${String.fromCharCode(65+idx)}. ${opt}`;

optionDiv.addEventListener('click', () => {

if (trainingAnswers[trainingIndex] === idx) {

trainingAnswers[trainingIndex] = null;

} else {

trainingAnswers[trainingIndex] = idx;

}

renderTraining(); // re-render to update selection & feedback

});

trainingOptionsDiv.appendChild(optionDiv);

});

// feedback: show correct answer if user selected something

const selected = trainingAnswers[trainingIndex];

if (selected !== null) {

const isCorrect = (selected === q.correct);

const correctLetter = String.fromCharCode(65+q.correct);

const correctAnswerText = q.options[q.correct];

if (isCorrect) {

trainingFeedbackDiv.innerHTML = `<div class="feedback correct-feedback">✅ Correct! ${correctAnswerText} is the right answer.</div>`;

} else {

trainingFeedbackDiv.innerHTML = `<div class="feedback wrong-feedback">❌ Incorrect. The correct answer is ${correctLetter}. ${correctAnswerText}.<br> 📘 Learn: ${q.text}</div>`;

}

} else {

trainingFeedbackDiv.innerHTML = `<div class="feedback">💡 Select an answer to see instant feedback (training mode).</div>`;

}

}

function nextTraining() {

if (trainingIndex + 1 < QUESTION_BANK.length) {

trainingIndex++;

renderTraining();

} else {

// wrap around or show summary

if (trainingAnswers.every(a => a !== null)) {

trainingFeedbackDiv.innerHTML = `<div class="feedback correct-feedback">🎉 Great! You've reviewed all 30 questions. Use exam mode to test yourself!</div>`;

} else {

trainingFeedbackDiv.innerHTML = `<div class="feedback">📚 You reached the end. Keep practicing — you can go back/forth or restart by toggling mode. Use 'Prev'? (click Next again to restart)</div>`;

if (trainingIndex === QUESTION_BANK.length-1) {

// allow wrap if needed but we keep, or just restart from 0 optional

if(confirm("You completed all 30 training questions. Restart from Q1?")) {

trainingIndex = 0;

renderTraining();

}

}

}

if (trainingIndex === QUESTION_BANK.length-1 && trainingAnswers[trainingIndex] !== null) {

// just stay, no auto wrap

}

}

}

// ----- EXAM logic -----

function stopExamTimer() {

if (examTimer) {

clearInterval(examTimer);

examTimer = null;

}

}

function startExamTimer() {

if (examTimer) stopExamTimer();

examTimer = setInterval(() => {

if (examSubmitted) return;

if (examTimeSeconds <= 0) {

// time's up -> auto submit

clearInterval(examTimer);

examTimer = null;

alert("⏰ Time's up! Submitting your exam automatically.");

submitExam();

} else {

examTimeSeconds--;

const minutes = Math.floor(examTimeSeconds / 60);

const secs = examTimeSeconds % 60;

examProgressSpan.innerText = `📋 Question ${examIndex+1} / 30 | ⏱️ Time: ${minutes.toString().padStart(2,'0')}:${secs.toString().padStart(2,'0')}`;

}

}, 1000);

}

function renderExam() {

if (examSubmitted) return;

const q = QUESTION_BANK[examIndex];

const minutesLeft = Math.floor(examTimeSeconds / 60);

const secsLeft = examTimeSeconds % 60;

examProgressSpan.innerText = `📋 Question ${examIndex+1} / 30 | ⏱️ Time: ${minutesLeft.toString().padStart(2,'0')}:${secsLeft.toString().padStart(2,'0')}`;

examQuestionDiv.innerText = q.text;

examOptionsDiv.innerHTML = '';

q.options.forEach((opt, idx) => {

const optDiv = document.createElement('div');

optDiv.className = 'option';

if (examAnswers[examIndex] === idx) optDiv.classList.add('selected');

optDiv.innerText = `${String.fromCharCode(65+idx)}. ${opt}`;

optDiv.addEventListener('click', () => {

if (examSubmitted) return;

if (examAnswers[examIndex] === idx) {

examAnswers[examIndex] = null;

} else {

examAnswers[examIndex] = idx;

}

renderExam(); // refresh selection, remove previous feedback if any

examFeedbackDiv.innerHTML = ''; // clear feedback on new selection

});

examOptionsDiv.appendChild(optDiv);

});

// no automatic feedback in exam mode until after submission (or next? but keep clean)

examFeedbackDiv.innerHTML = '';

// show submit button if it's the last question and we have answer optional

if (examIndex === QUESTION_BANK.length-1) {

examNextBtn.style.display = "none";

submitExamBtn.style.display = "inline-flex";

} else {

examNextBtn.style.display = "inline-flex";

submitExamBtn.style.display = "none";

}

}

function nextExam() {

if (examSubmitted) return;

// check if current question has an answer selected, if not warn

if (examAnswers[examIndex] === null) {

examFeedbackDiv.innerHTML = `<div class="feedback wrong-feedback">⚠️ Please select an answer before moving to the next question.</div>`;

return;

}

if (examIndex + 1 < QUESTION_BANK.length) {

examIndex++;

renderExam();

} else {

// last question, trigger submit if all answered? but they can also press submit button.

if (examAnswers[examIndex] !== null && examIndex === QUESTION_BANK.length-1) {

// offer submit

const confirmSubmit = confirm("You are at the last question. Do you want to submit your exam now?");

if (confirmSubmit) submitExam();

} else {

renderExam();

}

}

}

function submitExam() {

if (examSubmitted) return;

stopExamTimer();

// check all questions answered

let unanswered = [];

for (let i=0; i<QUESTION_BANK.length; i++) {

if (examAnswers[i] === null) unanswered.push(i+1);

}

let finalMessage = "";

if (unanswered.length > 0) {

finalMessage = `⚠️ You left ${unanswered.length} question(s) unanswered (Q${unanswered.join(", ")}). The exam will be marked as is.\n\n`;

}

// calculate score

let correctCount = 0;

for (let i=0; i<QUESTION_BANK.length; i++) {

if (examAnswers[i] !== null && examAnswers[i] === QUESTION_BANK[i].correct) correctCount++;

}

const percentage = (correctCount / QUESTION_BANK.length) * 100;

const passMark = 75; // UK Level 2 standard pass mark (75%)

const isPass = percentage >= passMark;

let resultHtml = `<div class="result-box"><strong>📊 Exam result</strong><br>`;

resultHtml += `✅ Correct: ${correctCount} / 30 (${percentage.toFixed(1)}%)<br>`;

resultHtml += `🎯 Pass mark: 75%<br>`;

if (isPass) {

resultHtml += `<span class="pass">✨ PASS ✨ You have met the Level 2 standard.</span><br>`;

} else {

resultHtml += `<span class="fail">❌ FAIL ❌ Please review training mode and retake the exam.</span><br>`;

}

resultHtml += `<button id="restartExamBtn" class="secondary" style="margin-top:15px;">🔄 Retake exam</button>`;

resultHtml += `<button id="gotoTrainingBtn" style="margin-left:10px;">📘 Go to training mode</button></div>`;

examFeedbackDiv.innerHTML = resultHtml;

examSubmitted = true;

examNextBtn.disabled = true;

submitExamBtn.disabled = true;

examNextBtn.style.display = "none";

submitExamBtn.style.display = "none";

// attach event handlers for result buttons

document.getElementById('restartExamBtn')?.addEventListener('click', () => {

resetExam();

});

document.getElementById('gotoTrainingBtn')?.addEventListener('click', () => {

setMode('training');

});

}

function resetExam() {

stopExamTimer();

examIndex = 0;

examAnswers = new Array(QUESTION_BANK.length).fill(null);

examSubmitted = false;

examTimeSeconds = 18 * 60;

examNextBtn.disabled = false;

submitExamBtn.disabled = false;

renderExam();

startExamTimer();

examFeedbackDiv.innerHTML = '';

}

// ---- Mode switching ----

function setMode(mode) {

currentMode = mode;

if (mode === 'training') {

trainingSection.style.display = 'block';

examSection.style.display = 'none';

trainingModeBtn.classList.add('active');

examModeBtn.classList.remove('active');

// stop any exam timer

stopExamTimer();

renderTraining();

} else {

trainingSection.style.display = 'none';

examSection.style.display = 'block';

trainingModeBtn.classList.remove('active');

examModeBtn.classList.add('active');

// reset or keep exam state? Better reset exam fresh for fairness

if (examSubmitted || examAnswers.some(a => a !== null)) {

// fresh start when entering exam mode

examIndex = 0;

examAnswers = new Array(QUESTION_BANK.length).fill(null);

examSubmitted = false;

examTimeSeconds = 18 * 60;

if (examTimer) stopExamTimer();

renderExam();

startExamTimer();

examFeedbackDiv.innerHTML = '';

} else {

if (!examTimer) startExamTimer();

renderExam();

}

}

}

// event binding

trainingNextBtn.addEventListener('click', nextTraining);

examNextBtn.addEventListener('click', nextExam);

submitExamBtn.addEventListener('click', submitExam);

trainingModeBtn.addEventListener('click', () => setMode('training'));

examModeBtn.addEventListener('click', () => setMode('exam'));

// initialise training

trainingIndex = 0;

trainingAnswers.fill(null);

renderTraining();

// Preload exam in background (not active) but ensure it's ready

resetExam(); // initial exam ready

setMode('training'); // default training mode active

</script>

</body>

</html>
